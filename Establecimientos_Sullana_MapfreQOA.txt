SELECT 
TERR.C_DES_ZONAL AS TERRITORIO ,
ZON.C_DES_ZONAL AS ZONA, 
REG.C_DES_ZONAL AS REGION ,
PVT.C_DES_ZONAL AS ESTABLECIMIENTO , COUNT(*)


FROM T_MAE_CERTIFICADO CER 
LEFT JOIN T_MAE_POLIZA  POL ON CER.N_ID_POLIZA = POL.N_ID_POLIZA
LEFT JOIN T_MAE_ESTABLECIMIENTO EST ON CER.N_ID_ESTABLECIMIENTO = EST.N_ID_ESTABLECIMIENTO
LEFT JOIN T_MAE_ZONAL PVT ON EST.N_ID_ZONAL = PVT.N_ID_ZONAL 
LEFT JOIN T_MAE_ZONAL REG ON PVT.N_ID_ZONALPADRE = REG.N_ID_ZONAL -- REGION 
LEFT JOIN T_MAE_ZONAL ZON ON REG.N_ID_ZONALPADRE = ZON.N_ID_ZONAL -- ZONA 
LEFT JOIN T_MAE_ZONAL TERR ON ZON.N_ID_ZONALPADRE = TERR.N_ID_ZONAL --TERRITORIO 
LEFT JOIN T_MAE_ENTIDAD ENTI ON TERR.N_ID_ENTIDAD= ENTI.N_ID_ENTIDAD
WHERE CER.N_ID_CANAL=7
GROUP BY REG.C_DES_ZONAL, ZON.C_DES_ZONAL,TERR.C_DES_ZONAL,PVT.C_DES_ZONAL
order by 5 desc

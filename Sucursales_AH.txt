SELECT 
PV.NOMBRE AS PUNTO_DE_VENTA, 
SU.NOMBRE AS SUCURSAL,
CA.NOMBRE AS NOMBRE_CANAL
FROM TB_ADM_PUNTO_VENTA PV
JOIN TB_ADM_SUCURSAL SU WITH(NOLOCK) ON PV.SUCURSAL_ID = SU.SUCURSAL_ID
JOIN TB_ADM_CANAL CA WITH(NOLOCK) ON SU.CANAL_ID= CA.CANAL_ID
WHERE SU.CANAL_ID IN (7,10,20)
ORDER BY 3 

SELECT * FROM TB_ADM_CANAL WHERE CANAL_ID IN (7,10, 20)

SELECT * FROM T_MAE_ENTIDAD
SELECT * FROM T_MAE_USUARIO_CANAL_PTOVENTA
